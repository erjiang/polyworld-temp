if [ $CONFIGURATION == "Release" ] ; then
	if [ ! -e $CONTENTS_FOLDER_PATH/Frameworks ] ; then
		mkdir $CONTENTS_FOLDER_PATH/Frameworks ;
	fi
	if [ ! -e $CONTENTS_FOLDER_PATH/Frameworks/QtCore ] ; then
		cp $QtLib/QtCore.framework/Versions/Current/QtCore $CONTENTS_FOLDER_PATH/Frameworks/ ;
	fi
	if [ ! -e $CONTENTS_FOLDER_PATH/Frameworks/QtGui.framework ] ; then
		cp $QtLib/QtGui.framework/Versions/Current/QtGui $CONTENTS_FOLDER_PATH/Frameworks/ ;
	fi
	if [ ! -e $CONTENTS_FOLDER_PATH/Frameworks/QtOpenGL.framework ] ; then
		cp $QtLib/QtOpenGL.framework/Versions/Current/QtOpenGL $CONTENTS_FOLDER_PATH/Frameworks/ ;
	fi
fi
